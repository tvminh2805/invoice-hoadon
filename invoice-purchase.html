function doGet(e) { try { const maDon = e.parameter.maDon || ''; if (!maDon) throw new Error('Thiếu mã đơn hàng'); // === CHỈ ĐỊNH RÕ FILE GOOGLE SHEETS === const SHEET_ID = '1A2B3C4D5E6F7G8H9I0J'; // ← THAY BẰNG ID CỦA BẠN const ss = SpreadsheetApp.openById(SHEET_ID); // === LẤY DỮ LIỆU TỪ 3 SHEET === const sheetDH_NCC = ss.getSheetByName('DH_NCC'); const sheetDHCT_NCC = ss.getSheetByName('DHCT_NCC'); const sheetThanhToan_NCC = ss.getSheetByName('ThanhToan_NCC'); if (!sheetDH_NCC) throw new Error('Không tìm thấy sheet "DH_NCC"'); if (!sheetDHCT_NCC) throw new Error('Không tìm thấy sheet "DHCT_NCC"'); if (!sheetThanhToan_NCC) throw new Error('Không tìm thấy sheet "ThanhToan_NCC"'); // === 1. LẤY THÔNG TIN CHÍNH === const donHangData = sheetDH_NCC.getDataRange().getValues(); const headerDon = donHangData[0]; const idDHIndex = headerDon.indexOf('IDDH_NCC'); const ngayNhapIndex = headerDon.indexOf('NgayNhap'); const tenNCCIndex = headerDon.indexOf('TenNCC'); const loaiGDIndex = headerDon.indexOf('LoaiGD'); let donHang = null; for (let i = 1; i < donHangData.length; i++) { if (String(donHangData[i][idDHIndex]) == maDon) { donHang = donHangData[i]; break; } } if (!donHang) throw new Error('Không tìm thấy đơn hàng #' + maDon); // === 2. LẤY CHI TIẾT ĐƠN HÀNG === const chiTietData = sheetDHCT_NCC.getDataRange().getValues(); const headerCT = chiTietData[0]; const idDHCTIndex = headerCT.indexOf('IDDH_NCC'); const tenHHIndex = headerCT.indexOf('TenHH'); const dvtIndex = headerCT.indexOf('DVT'); const slNhapIndex = headerCT.indexOf('SLNhap'); const giaNhapIndex = headerCT.indexOf('GiaNhap'); const thanhTienIndex = headerCT.indexOf('ThanhTien'); const chiTiet = []; for (let i = 1; i < chiTietData.length; i++) { if (String(chiTietData[i][idDHCTIndex]) == maDon) { chiTiet.push({ tenSanPham: String(chiTietData[i][tenHHIndex]) || '', dvt: String(chiTietData[i][dvtIndex]) || '', soLuong: Number(chiTietData[i][slNhapIndex]) || 0, donGia: Number(chiTietData[i][giaNhapIndex]) || 0, thanhTien: Number(chiTietData[i][thanhTienIndex]) || 0 }); } } // === 3. LẤY THÔNG TIN THANH TOÁN (ƯU TIÊN) === const thanhToanData = sheetThanhToan_NCC.getDataRange().getValues(); const headerTT = thanhToanData[0]; const idDHTTIndex = headerTT.indexOf('IDDH_NCC'); const ngayTTIndex = headerTT.indexOf('Ngay'); // Lấy ngày từ ThanhToan_NCC const tenNCCCTIndex = headerTT.indexOf('TenNCC'); const loaiGDTTIndex = headerTT.indexOf('LoaiGD'); const tongTienHangIndex = headerTT.indexOf('TongTienHang'); const noCuIndex = headerTT.indexOf('NoCu'); const tongThanhToanIndex = headerTT.indexOf('TongThanhToan'); const tienDuaNCCIndex = headerTT.indexOf('TienDuaNCC'); const tienThuaNoIndex = headerTT.indexOf('TienThua_No'); let thanhToan = null; for (let i = 1; i < thanhToanData.length; i++) { if (String(thanhToanData[i][idDHTTIndex]) == maDon) { thanhToan = thanhToanData[i]; break; } } // === XỬ LÝ DỮ LIỆU THANH TOÁN === // Nếu chưa có dữ liệu thanh toán, lấy từ sheet DH_NCC if (!thanhToan) { // Tính tổng tiền hàng từ chi tiết let tongTienHang = 0; chiTiet.forEach(item => tongTienHang += item.thanhTien); thanhToan = { ngay: String(donHang[ngayNhapIndex]) || '', // Lấy ngày nhập từ DH_NCC tenNCC: String(donHang[tenNCCIndex]) || 'Nhà cung cấp', loaiGD: String(donHang[loaiGDIndex]) || 'Mua hàng', tongTienHang: tongTienHang, noCu: 0, tongThanhToan: tongTienHang, tienDuaNCC: 0, tienThuaNo: -tongTienHang // Còn nợ nhà cung cấp }; } else { // Ưu tiên lấy thông tin từ sheet ThanhToan_NCC thanhToan = { ngay: String(thanhToan[ngayTTIndex]) || '', // Lấy ngày từ ThanhToan_NCC tenNCC: String(thanhToan[tenNCCCTIndex]) || 'Nhà cung cấp', loaiGD: String(thanhToan[loaiGDTTIndex]) || 'Mua hàng', tongTienHang: Number(thanhToan[tongTienHangIndex]) || 0, noCu: Number(thanhToan[noCuIndex]) || 0, tongThanhToan: Number(thanhToan[tongThanhToanIndex]) || 0, tienDuaNCC: Number(thanhToan[tienDuaNCCIndex]) || 0, tienThuaNo: Number(thanhToan[tienThuaNoIndex]) || 0 }; } // === 4. TRẢ VỀ JSON === const result = { success: true, maDon: maDon, ngay: thanhToan.ngay, // Ngày thanh toán (ưu tiên từ ThanhToan_NCC) tenNCC: thanhToan.tenNCC, loaiGD: thanhToan.loaiGD, chiTiet: chiTiet, thanhToan: thanhToan }; return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.message })).setMimeType(ContentService.MimeType.JSON); } }
